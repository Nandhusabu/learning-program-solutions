class student {
    private String name;
    private String id;
    private String grade;

    public student(String name, String id, String grade) {
        this.name = name;
        this.id = id;
        this.grade = grade;
    }

    public String getname() {
        return name;
    }

    public void setname(String name) {
        this.name = name;
    }

    public String getid() {
        return id;
    }

    public void setid(String id) {
        this.id = id;
    }

    public String getgrade() {
        return grade;
    }

    public void setgrade(String grade) {
        this.grade = grade;
    }
}

class studentview {
    public void showstudent(String name, String id, String grade) {
        System.out.println("Student Details:");
        System.out.println("Name: " + name);
        System.out.println("ID: " + id);
        System.out.println("Grade: " + grade);
    }
}

class studentcontroller {
    private student s;
    private studentview v;

    public studentcontroller(student s, studentview v) {
        this.s = s;
        this.v = v;
    }

    public void setname(String name) {
        s.setname(name);
    }

    public void setid(String id) {
        s.setid(id);
    }

    public void setgrade(String grade) {
        s.setgrade(grade);
    }

    public void updatestudent() {
        v.showstudent(s.getname(), s.getid(), s.getgrade());
    }
}

public class mvcapp {
	    public static void main(String[] args) {
	        student stu = new student("ARJUN", "S105", "A");
	        studentview view = new studentview();
	        studentController ctrl = new studentController(stu, view);

	        ctrl.updatestudent();

	        ctrl.setname("arvind");
	        ctrl.setgrade("A+");

	        ctrl.updatestudent();
	    }
	}
